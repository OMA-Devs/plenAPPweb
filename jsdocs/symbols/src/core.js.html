 42 var calls = {
 43 	/**
 44 	 * Llamada generica a archivos php del servidor. Es lo más flexible posible para
 45 	 * ser utilizada en los maximos sitios posibles.
 46 	 * @param {Array} keyARR Contiene las KEYS para generar el query HTTP
 47 	 * @param {Array} valARR Contiene los valores para generar el query HTTP
 48 	 * @param {String} fileTARGET Archivo php de destino en el servidor
 49 	 * @param {String} divTARGET ID del objetivo. Si se entrega una cadena vacía, la
 50 	 * respuesta se dará en un alert, y si no se modificará el elemento entregado.
 51 	 */
 52 	genericLoad: function(keyARR, valARR, fileTARGET, divTARGET){
 53 		coreUI.clearIntervals()
 54 		var xhttp = new XMLHttpRequest();
 55 		xhttp.onreadystatechange = function() {
 56 		if (this.readyState == 4 && this.status == 200) {
 57 				if (divTARGET == ""){
 58 					console.log(this.responseText)
 59 					alert(this.responseText)
 60 				}else{
 61 					var view = document.getElementById(divTARGET);
 62 					view.innerHTML = ""
 63 					view.innerHTML = this.responseText
 64 				}
 65 			}
 66 		};
 67 		var query = fileTARGET+"?"
 68 		for (var i = 0; i<keyARR.length; i++){
 69 			if (i == keyARR.length-1){
 70 				query += keyARR[i]+"="+valARR[i]
 71 			}else{
 72 				query += keyARR[i]+"="+valARR[i]+"&"
 73 			}
 74 		}
 75 		console.log(query)
 76 		xhttp.open("GET", query, true);
 77 		xhttp.send();
 78 	},
 79 	/**
 80 	 * Funcion de Login. Funcionalidad muy básica y seguridad nula. Debe mejorarse.
 81 	 * Modifica la variable global userConnected si los datos entregados son correctos
 82 	 * en la base de datos.
 83 	 * @see userConnected
 84 	 */
 85 	login: function(){
 86 		var user = document.getElementById("usr").value
 87 		var pass = document.getElementById("pass").value
 88 		var xhttp = new XMLHttpRequest();
 89 		xhttp.onreadystatechange = function() {
 90 		if (this.readyState == 4 && this.status == 200) {
 91 			//console.log(this.responseText)
 92 			var resp = JSON.parse(this.responseText)
 93 				if(resp.resp == "Login Correcto"){
 94 					//console.log(resp.usuario)
 95 					alert("Login Correcto")
 96 					userConnected = resp.usuario
 97 					if (userConnected.PERMadmin == "1"){
 98 						document.getElementById("administracion").style = "display:inline-block"
 99 					}
100 					if (userConnected.PERMhistoricos == "1"){
101 						document.getElementById("historicos").style = "display:inline-block"
102 					}
103 					if (userConnected.PERMincidencias == "1"){
104 						document.getElementById("incidencias").style = "display:inline-block"
105 					}
106 					document.getElementById("user").style = "display:inline-block"
107 					document.getElementById("logout").style = "display:inline-block"
108 					document.getElementById("content").innerHTML = ""
109 				}else{
110 					//console.log(this.responseText)
111 					alert("Login Incorrecto")
112 				}
113 			}
114 		};
115 		var query = "auth.php?type=auth"
116 		query += "&usr="+user
117 		query += "&pass="+pass
118 		console.log(query)
119 		xhttp.open("GET", query, true);
120 		xhttp.send();
121 	},
122 	/**
123 	 * Función de Logout. Tanto o mas simple que el login. Debe mejorarse.
124 	 * Modifica la funcion userConnected.
125 	 * @see userConnected
126 	 */
127 	logout: function(){
128 		userConnected = undefined
129 		var bar = document.getElementById("navBar")
130 		for(var i = 0; i<bar.children.length; i++){
131 			bar.children[i].style = "display: none"
132 		}
133 		coreUI.clearIntervals()
134 		calls.genericLoad(['type'],['show'],'auth.php','content')
135 	}

144 	 * Almacena el nodo entregado como argumento en la variable selectedRow.

161 		}

338 var coreACTIONS = {
339 	editTable : function(element, target){
340 		if (edit == true && delRow == true){
341 			alert("Edicion y eliminado estan activados a la vez. Desactiva uno")
342 		} else{
343 			if (edit == true){
344 				var newVal = prompt("Introduzca nuevo valor:", element.innerHTML)
345 				element.innerHTML = newVal
346 				//Obtenemos los headers de la tabla
347 				var tab = document.getElementById("activeTable")
348 				var head = tab.children[0].children[0]
349 				keyARR = ["type"]
350 				for (var i = 0; i<head.children.length; i++){
351 					//console.log(head.children[i].innerHTML)
352 					//Es importante convertir a minusculas los campos.
353 					keyARR.push(head.children[i].innerHTML.toLowerCase())
354 				}
355 				//Obtenemos los valores de la linea a modificar
356 				valARR = ["edit"]
357 				for (var i = 0; i<element.parentElement.children.length; i++){
358 					valARR.push(element.parentElement.children[i].innerHTML)
359 				}
360 				//Mandamos los datos al servidor.
361 				calls.genericLoad(keyARR,valARR,target,"")
362 			}
363 			if (delRow == true){
364 				if (selectedRow == undefined){
365 					alert("Selecciona una fila para eliminar")
366 				}else{
367 					var conf = confirm("¿Eliminar esta entrada?")
368 					if (conf == true){
369 						var idRow = selectedRow.children[0].innerHTML
370 						calls.genericLoad(["type","id"],["delete",idRow],target,"")
371 						calls.genericLoad(["type"],["show"],target,"content")
372 						coreUI.resetGlobals()
373 					}
374 				}
375 			}
376 		}
377 	},
378 	addRow : function(target){
379 		//Obtenemos los headers de la tabla
380 		var tab = document.getElementById("activeTable")
381 		var head = tab.children[0].children[0]
382 		keyARR = []
383 		for (var i = 0; i<head.children.length; i++){
384 			if (head.children[i].innerHTML.toLowerCase() == "id" || head.children[i].innerHTML.toLowerCase() == "copia"){
385 				continue
386 			}else{
387 				keyARR.push(head.children[i].innerHTML.toLowerCase())
388 			}
389 		}
390 		//Solicitamos un valor por cada celda excepto para ID que es AUTOINCREMENT en la base de datos
391 		valARR = []
392 		for (var i = 0; i<keyARR.length; i++){
393 			var newVal = prompt("Introduce valor para "+keyARR[i], keyARR[i])
394 			valARR.push(newVal)
395 		}
396 		keyARR.unshift("type")
397 		valARR.unshift("add")
398 		calls.genericLoad(keyARR,valARR,target,"")
399 		calls.genericLoad(["type"],["show"],target,"content")
400 		coreUI.resetGlobals()
401 	},
