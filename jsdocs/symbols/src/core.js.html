 28 /** Incluye todas las llamadas a servidor con sus respectivos argumentos.
 29  * @namespace
 30  */
 31 var calls = {
 32 	/**
 33 	 * Llamada generica a archivos php del servidor. Es lo más flexible posible para
 34 	 * ser utilizada en los maximos sitios posibles.
 35 	 * @param {Array} keyARR Contiene las KEYS para generar el query HTTP
 36 	 * @param {Array} valARR Contiene los valores para generar el query HTTP
 37 	 * @param {String} fileTARGET Archivo php de destino en el servidor
 38 	 * @param {String} divTARGET ID del objetivo. Si se entrega una cadena vacía, la
 39 	 * respuesta se dará en un alert, y si no se modificará el elemento entregado.
 40 	 */
 41 	genericLoad: function(keyARR, valARR, fileTARGET, divTARGET){
 42 		var xhttp = new XMLHttpRequest();
 43 		xhttp.onreadystatechange = function() {
 44 		if (this.readyState == 4 && this.status == 200) {
 45 				if (divTARGET == ""){
 46 					console.log(this.responseText)
 47 					alert(this.responseText)
 48 				}else{
 49 					var view = document.getElementById(divTARGET);
 50 					view.innerHTML = ""
 51 					view.innerHTML = this.responseText
 52 				}
 53 			}
 54 		};
 55 		var query = fileTARGET+"?"
 56 		for (var i = 0; i<keyARR.length; i++){
 57 			if (i == keyARR.length-1){
 58 				query += keyARR[i]+"="+valARR[i]
 59 			}else{
 60 				query += keyARR[i]+"="+valARR[i]+"&"
 61 			}
 62 		}
 63 		console.log(query)
 64 		xhttp.open("GET", query, true);
 65 		xhttp.send();
 66 	},
 67 	/**
 68 	 * Funcion de Login. Funcionalidad muy básica y seguridad nula. Debe mejorarse.
 69 	 * Modifica la variable global userConnected si los datos entregados son correctos
 70 	 * en la base de datos.
 71 	 * @see userConnected
 72 	 */
 73 	login: function(){
 74 		var user = prompt("Introduzca usuario:","usuario")
 75 		var pass = prompt("Introduzca contraseña:","contraseña")
 76 		var xhttp = new XMLHttpRequest();
 77 		xhttp.onreadystatechange = function() {
 78 		if (this.readyState == 4 && this.status == 200) {
 79 				if(this.responseText == "Login Correcto"){
 80 					//console.log(this.responseText)
 81 					alert("Login Correcto")
 82 					userConnected = user
 83 					var bar = document.getElementById("navBar")
 84 					for(var i = 0; i<bar.children.length; i++){
 85 						bar.children[i].style = "display: inline-block"
 86 					}
 87 				}else{
 88 					//console.log(this.responseText)
 89 					alert("Login Incorrecto")
 90 				}
 91 			}
 92 		};
 93 		var query = "auth.php?"
 94 		query += "usr="+user
 95 		query += "&pass="+pass
 96 		//console.log(query)
 97 		xhttp.open("GET", query, true);
 98 		xhttp.send();
 99 	},
100 	/**
101 	 * Función de Logout. Tanto o mas simple que el login. Debe mejorarse.
102 	 * Modifica la funcion userConnected.
103 	 * @see userConnected
104 	 */
105 	logout: function(){
106 		userConnected = undefined
107 		var bar = document.getElementById("navBar")
108 		for(var i = 1; i<bar.children.length; i++){
109 			bar.children[i].style = "display: none"
110 		}
111 	}
112 }

114 /** Incluye todas las funciones gráficas de UI y modifica variables globales relacionadas
115  * con ello.
116  * @namespace
117  */
118 var coreUI = {
119 	/**
120 	 * Almacena el nodo entregado como argumento en la variable selectedRow.
121 	 * Además, cambia el color de fondo del nodo para distinguir la seleccion.
122 	 * @param {HTMLElement} node Nodo elegido, normalmente por click en la interfaz.
123 	 * @see selectedRow
124 	 */
125 	getValue : function(node){
126 		//console.log(node.parentElement.children[0])
127 		if (selectedRow == undefined){
128 			selectedRow = node
129 			selectedRow.style = "background-color: yellow"
130 		}else if(selectedRow == node){
131 			selectedRow.style = ""
132 			selectedRow = undefined
133 		}else{
134 			selectedRow.style = ""
135 			selectedRow = node
136 			selectedRow.style = "background-color: yellow"
137 		}
138 	},
139 	/**
140 	 * Funcion de conveniencia para resetear las variables. Utilizada al cargar secciones de
141 	 * la interfaz para evitar bugs. Por ejemplo, modificar un campo que no esta ya en pantalla
142 	 * pero programaticamente marcado como seleccionado.
143 	 * @see selectedRow
144 	 * @see edit
145 	 * @see delRow
146 	 */
147 	resetGlobals : function(){
148 		selectedRow = false
149 		edit = false
150 		delRow = false
151 	},
152 	/**
153 	 * Modifica el color del nodo entregado para determinar visualmente que la edicion de campos
154 	 * esta activa. Ademas activa/desactiva la variable global edit.
155 	 * @param {HTMLElement} node Nodo que activa y desactiva la opcion en cuestion.
156 	 * @see edit
157 	 */
158 	toogleEdit : function(node){
159 		if (edit == true){
160 			node.style = ""
161 			edit = false
162 		}else{
163 			node.style = "background-color: green"
164 			edit = true
165 		}
166 	},
167 	/**
168 	 * Modifica el color del nodo entregado para determinar visualmente que la eliminación de
169 	 * filas esta activa. Ademas activa/desactiva la variable global delRow
170 	 * @param {HTMLElement} node Nodo que activa y desactiva la opcion en cuestion.
171 	 * @see delRow
172 	 */
173 	toogleDelete : function(node){
174 		if (delRow == true){
175 			node.style = ""
176 			delRow = false
177 		}else{
178 			node.style = "background-color: red"
179 			delRow = true
180 		}
181 	}
182 }

184 /**
185  * Acciones dedicadas a una seccion en concreto, con llamadas especificas al servidor y gestiones específicas de UI.
186  * @namespace
187  */
188 var coreACTIONS = {
189 	/**
190 	 * Función para editar y eliminar objetos de la interfaz relacionada con "ESTACIONES". Contiene la funcionalidad
191 	 * de seguridad necesaria para no tener el modo de edicion y eliminacion conjuntamente, y efectua la funcion
192 	 * necesaria dependiendo de las variables activas.
193 	 * @param {HTMLElement} element Campo pulsado especificamente. Se utiliza en el modo edicion. En el modo
194 	 * eliminacion se utiliza para determinar la fila padre a eliminar.
195 	 * @see edit
196 	 * @see delRow
197 	 */
198 	editEstacion : function(element){
199 		if (edit == true && delRow == true){
200 			alert("Edicion y eliminado estan activados a la vez. Desactiva uno")
201 		} else{
202 			if (edit == true){
203 				var newVal = prompt("Introduzca nuevo valor:", element.innerHTML)
204 				element.innerHTML = newVal
205 				//Obtenemos los headers de la tabla
206 				var tab = document.getElementById("activeTable")
207 				var head = tab.children[0].children[0]
208 				keyARR = ["type"]
209 				for (var i = 0; i<head.children.length; i++){
210 					//console.log(head.children[i].innerHTML)
211 					//Es importante convertir a minusculas los campos.
212 					keyARR.push(head.children[i].innerHTML.toLowerCase())
213 				}
214 				//Obtenemos los valores de la linea a modificar
215 				valARR = ["edit"]
216 				for (var i = 0; i<element.parentElement.children.length; i++){
217 					valARR.push(element.parentElement.children[i].innerHTML)
218 				}
219 				//Mandamos los datos al servidor.
220 				calls.genericLoad(keyARR,valARR,"editEstaciones.php","")
221 			}
222 			if (delRow == true){
223 				if (selectedRow == undefined){
224 					alert("Selecciona una fila para eliminar")
225 				}else{
226 					var conf = confirm("¿Eliminar esta estacion?")
227 					if (conf == true){
228 						var idRow = selectedRow.children[0].innerHTML
229 						calls.genericLoad(["type","id"],["delete",idRow],"estaciones.php","")
230 						calls.genericLoad(["type"],["show"],"estaciones.php","content")
231 					}
232 				}
233 			}
234 		}
235 	},
236 	/**
237 	 * Funcion para añadir estaciones desde la interfaz grafica. Contiene la llamada concreta al servidor.
238 	 */
239 	addEstacion : function(){
240 		//Obtenemos los headers de la tabla
241 		var tab = document.getElementById("activeTable")
242 		var head = tab.children[0].children[0]
243 		keyARR = []
244 		for (var i = 0; i<head.children.length; i++){
245 			if (head.children[i].innerHTML.toLowerCase() == "id" || head.children[i].innerHTML.toLowerCase() == "copia"){
246 				continue
247 			}else{
248 				keyARR.push(head.children[i].innerHTML.toLowerCase())
249 			}
250 		}
251 		//Solicitamos un valor por cada celda excepto para ID que es AUTOINCREMENT en la base de datos
252 		valARR = []
253 		for (var i = 0; i<keyARR.length; i++){
254 			var newVal = prompt("Introduce valor para "+keyARR[i], keyARR[i])
255 			valARR.push(newVal)
256 		}
257 		keyARR.unshift("type")
258 		valARR.unshift("add")
259 		calls.genericLoad(keyARR,valARR,"estaciones.php","")
260 		calls.genericLoad(["type"],["show"],"estaciones.php","content")
261 	}
262 }
