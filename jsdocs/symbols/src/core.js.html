 93 			}

 99 			}else{

101 			}
102 		}
103 		console.log(query)
104 		xhttp.open("GET", query, true);
105 		xhttp.send();

544 				var tab = document.getElementById("activeTable")

551 				}
552 				//Obtenemos los valores de la linea a modificar
553 				valARR = ["edit"]
554 				for (var i = 0; i<element.parentElement.children.length; i++){
555 					valARR.push(element.parentElement.children[i].innerHTML)
556 				}
557 				//Mandamos los datos al servidor.
558 				calls.genericLoad(keyARR,valARR,target,"")
559 				coreUI.resetGlobals()
560 				if (target == "presets.php"){
561 					calls.getPresets()
562 				}
563 			}
564 			if (delRow == true){
565 				if (selectedRow == undefined){
566 					alert("Selecciona una fila para eliminar")
567 				}else{
568 					var conf = confirm("¿Eliminar esta entrada?")
569 					if (conf == true){
570 						var idRow = selectedRow.children[0].innerHTML

574 						if (target == "presets.php"){
